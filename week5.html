Step 1: Matplotlib 준비하기

import matplotlib.pyplot as plt
import numpy as np

# 한글 폰트 설정 (필요시)
plt.rcParams['font.family'] = 'Malgun Gothic'

Step 2: 내 일주일 기분 그래프

# 일주일 동안의 내 기분 점수 (1-10점)
days = ['월', '화', '수', '목', '금', '토', '일']
mood_scores = [__, __, __, __, __, __, __]  # 실제 점수 입력

# 선 그래프 그리기
plt.figure(figsize=(10, 6))
plt.plot(days, mood_scores, marker='o', linewidth=2, markersize=8)
plt.title('내 일주일 기분 변화')
plt.xlabel('요일')
plt.ylabel('기분 점수 (1-10)')
plt.grid(True)
plt.show()

  • 가장 좋았던 날: 요일 (점)
  • 가장 나빴던 날: 요일 (점)
  • 주말 vs 평일 기분:

Step 3: 그래프 꾸미기

# 더 예쁜 그래프 만들기
plt.figure(figsize=(10, 6))
plt.plot(days, mood_scores, 
         marker='o',           # 점 모양: 'o', 's', '^', '*' 등
         linewidth=3,          # 선 두께
         markersize=10,        # 점 크기
         color='___',          # 원하는 색상 입력
         linestyle='___')      # 선 스타일: '-', '--', '-.', ':' 등

plt.title('내 일주일 기분 변화 (꾸미기 버전)', fontsize=16)
plt.xlabel('요일', fontsize=12)
plt.ylabel('기분 점수', fontsize=12)
plt.grid(True, alpha=0.3)
plt.ylim(0, 10)  # y축 범위 설정
plt.show()

미션 1: 내 수면시간 그래프

# 일주일 수면시간 데이터
sleep_hours = [__, __, __, __, __, __, __]  # 실제 수면시간 입력

plt.figure(figsize=(10, 6))
plt.plot(days, sleep_hours, marker='s', linewidth=2, 
         markersize=8, color='blue')

# 권장 수면시간 선 추가
recommended_sleep = [8] * 7  # 8시간 권장
plt.plot(days, recommended_sleep, '--', color='red', 
         alpha=0.7, label='권장 수면시간 (8시간)')

plt.title('내 일주일 수면시간')
plt.xlabel('요일')
plt.ylabel('수면시간 (시간)')
plt.legend()  # 범례 표시
plt.grid(True, alpha=0.3)
plt.show()
😴 수면 패턴 분석:
  • 평균 수면시간: 시간
  • 권장 시간보다 부족한 날: 일
  • 개선 계획:

미션 2: 여러 선을 한 그래프에

# 공부시간과 게임시간 비교
study_hours = [__, __, __, __, __, __, __]
game_hours = [__, __, __, __, __, __, __]

plt.figure(figsize=(12, 6))
plt.plot(days, study_hours, 'o-', label='공부시간', 
         linewidth=2, markersize=8, color='green')
plt.plot(days, game_hours, 's-', label='게임시간', 
         linewidth=2, markersize=8, color='orange')

plt.title('공부시간 vs 게임시간')
plt.xlabel('요일')
plt.ylabel('시간')
plt.legend()
plt.grid(True, alpha=0.3)
plt.show()
⏰ 시간 사용 패턴 분석:
  • 공부 > 게임인 날: 일
  • 가장 균형잡힌 날: 요일
