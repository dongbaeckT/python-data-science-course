plt.show()

Step 3: 그라데이션과 투명도 효과

# 그라데이션 막대그래프
subjects = ['수학', '과학', '영어', '국어', '사회']
my_scores = [__, __, __, __, __]  # 내 점수 입력

# 점수에 따른 그라데이션 색상
def score_to_color(score):
    if score >= 90:
        return '#ff4757'  # 빨간색 (우수)
    elif score >= 80:  
        return '#ffa726'  # 주황색 (양호)
    elif score >= 70:
        return '#42a5f5'  # 파란색 (보통)
    else:
        return '#66bb6a'  # 초록색 (노력필요)

colors = [score_to_color(score) for score in my_scores]

plt.figure(figsize=(12, 8))

# 메인 막대그래프
bars = plt.bar(subjects, my_scores, color=colors, alpha=0.8, 
               edgecolor='white', linewidth=2)

# 그림자 효과
for i, (bar, score) in enumerate(zip(bars, my_scores)):
    # 그림자 막대 (약간 뒤쪽에)
    plt.bar(bar.get_x() + 0.02, score, bar.get_width(), 
             color='gray', alpha=0.3, zorder=0)
    
    # 점수 표시
    plt.text(bar.get_x() + bar.get_width()/2, score + 1, 
             f'{score}점', ha='center', va='bottom', 
             fontsize=12, fontweight='bold')

plt.title('내 성적 분석 (그라데이션 & 그림자 효과)', fontsize=16, fontweight='bold')
plt.xlabel('과목', fontsize=14)
plt.ylabel('점수', fontsize=14)
plt.ylim(0, 100)

# 범례 추가
from matplotlib.patches import Patch
legend_elements = [
    Patch(facecolor='#ff4757', label='우수 (90점 이상)'),
    Patch(facecolor='#ffa726', label='양호 (80-89점)'),
    Patch(facecolor='#42a5f5', label='보통 (70-79점)'),
    Patch(facecolor='#66bb6a', label='노력필요 (70점 미만)')
]
plt.legend(handles=legend_elements, loc='upper right')

plt.grid(True, axis='y', alpha=0.3)
plt.show()

# 시간에 따른 성장을 여러 프레임으로 보여주기

my_heights = [150, 155, 161, 165, 168, 170]  # 6년간 키 성장
ages = [10, 11, 12, 13, 14, 15]

fig, axes = plt.subplots(2, 3, figsize=(18, 10))
axes = axes.flatten()

for i in range(6):
    # 각 프레임마다 누적해서 보여주기
    current_ages = ages[:i+1]
    current_heights = my_heights[:i+1]
    
    axes[i].plot(current_ages, current_heights, 'o-', 
                 linewidth=3, markersize=10, color='#e74c3c')
    axes[i].set_xlim(9, 16)
    axes[i].set_ylim(145, 175)
    axes[i].set_title(f'{ages[i]}세까지의 성장', fontsize=14)
    axes[i].set_xlabel('나이')
    axes[i].set_ylabel('키 (cm)')
    axes[i].grid(True, alpha=0.3)
    
    # 현재 키 표시
    if current_heights:
        axes[i].text(current_ages[-1], current_heights[-1] + 2, 
                     f'{current_heights[-1]}cm', 
                     ha='center', fontweight='bold', fontsize=12)

plt.tight_layout()
plt.show()
