💻 실습 1: NumPy vs pandas - 무엇이 다를까?

NumPy로 성적 데이터 만들기

import numpy as np
# NumPy 배열로 성적표 만들기
print("=== NumPy 버전 성적표 ===")
scores_numpy = np.array([
[85, 92, 78], # 철수
[90, 88, 95], # 영희
[76, 82, 89] # 민수
])
print(scores_numpy)
print("이 숫자들이 뭘 의미하는지 알 수 있나요? 🤔")

pandas로 같은 데이터 만들기

import pandas as pd
# pandas DataFrame으로 성적표 만들기
print("\n=== pandas 버전 성적표 ===")
scores_pandas = pd.DataFrame({
'국어': [85, 90, 76],
'영어': [92, 88, 82],
'수학': [78, 95, 89]
}, index=['철수', '영희', '민수'])
print(scores_pandas)
print("어느 쪽이 더 이해하기 쉬운가요? ✨")

🎯 실습 3: DataFrame 조작의 기초

특정 데이터 가져오기

# 특정 컬럼 가져오기
print("=== 이름만 가져오기 ===")
names = friends_info['이름']
print(names)
print(f"데이터 타입: {type(names)}") # Series라고 나와요!
# 여러 컬럼 가져오기
print("\n=== 이름과 나이만 가져오기 ===")
name_age = friends_info[['이름', '나이']]
print(name_age)
print(f"데이터 타입: {type(name_age)}") # DataFrame이에요!
# 특정 행 가져오기
print("\n=== 첫 번째 친구 정보 ===")
first_friend = friends_info.iloc[0] # iloc: 위치로 접근
print(first_friend)
# 특정 데이터 하나만 가져오기
print(f"\n첫 번째 친구의 이름: {friends_info.iloc[0, 0]}")
print(f"영희의 취미: {friends_info.loc[1, '취미']}") # loc: 라벨로 접근

새로운 컬럼 추가하기

# 성별 컬럼 추가
friends_info['성별'] = ['남', '여', '남', '여']
# 가상의 몸무게 데이터 추가
friends_info['몸무게'] = [55, 48, 62, 45]
# BMI 계산해서 추가 (키는 cm 단위)
friends_info['BMI'] = friends_info['몸무게'] / ((friends_info['키'] / 100) ** 2)
print("=== 새 컬럼이 추가된 DataFrame ===")
print(friends_info)

# 일주일간 점심 메뉴와 만족도를 기록해보세요

# 게임 캐릭터들의 정보를 DataFrame으로 만들어보세요!
rpg_characters = pd.DataFrame({
'이름': ['용사 김용용', '마법사 이마마', '도적 박도도', '성직자 최성성'],
'직업': ['전사', '마법사', '도적', '성직자'],
'레벨': [25, 23, 27, 24],
'체력': [150, 80, 120, 100],
'마나': [50, 200, 70, 180],
'공격력': [80, 45, 75, 35],
'방어력': [70, 25, 40, 60]
})
print("=== RPG 파티 정보 ===")
print(rpg_characters)
# 파티 전체 능력 계산
total_hp = rpg_characters['체력'].sum()
total_attack = rpg_characters['공격력'].sum()

print(f"파티 총 공격력: {total_attack}")
# 가장 강한 캐릭터 찾기
strongest = rpg_characters.loc[rpg_characters['공격력'].idxmax()]
print(f"최강 공격력 캐릭터: {strongest['이름']} ({strongest['공격력']})")

💡 오늘 배운 것 정리

pandas의 장점

  • 가독성: 데이터가 표 형태로 깔끔하게 보임
  • 직관성: 엑셀처럼 친숙한 방식으로 데이터 조작
  • 편리성: 복잡한 데이터 분석이 간단한 코드로 가능
  • 실용성: 실제 데이터 과학에서 가장 많이 사용

핵심 개념

DataFrame: 표 형태의 2차원 데이터 구조

Series: 1차원 데이터 구조 (DataFrame의 한 컬럼)

index: 각 행을 구분하는 라벨

columns: 각 열의 이름

기본 메소드

.head(), .tail(): 데이터 미리보기

.info(), .describe(): 데이터 정보 확인

.iloc[], .loc[]: 특정 데이터 접근

.shape, .columns: DataFrame 구조 확인
