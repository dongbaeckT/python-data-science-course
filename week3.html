Step 1: 배열끼리 계산하기

import numpy as np

# 두 반의 시험 점수
class_a = np.array([85, 92, 78, 96, 88])
class_b = np.array([82, 89, 91, 93, 85])

# 점수 차이 계산
difference = class_a - class_b
print(f"A반과 B반의 점수 차이: {difference}")

# 평균 점수
average_score = (class_a + class_b) / 2
print(f"두 반 평균 점수: {average_score}")

# 모든 점수에 5점 보너스
bonus_a = class_a + 5
print(f"A반 보너스 후: {bonus_a}")

# 모든 점수를 2배로
double_b = class_b * 2
print(f"B반 2배: {double_b}")

  • 점수 차이: [ , , , , ]
  • 평균 점수: [ , , , , ]

Step 2: 유용한 통계 함수들

scores = np.array([72, 88, 91, 67, 94, 83, 79, 96, 85, 74])

print("=== 우리 반 점수 분석 ===")
print(f"최고 점수: {np.max(scores)}")
print(f"최저 점수: {np.min(scores)}")
print(f"평균 점수: {np.mean(scores):.1f}")
print(f"점수 합계: {np.sum(scores)}")
print(f"표준편차: {np.std(scores):.1f}")

# 점수들을 크기 순으로 정렬
sorted_scores = np.sort(scores)
print(f"점수 순서대로: {sorted_scores}")

# 어떤 인덱스 순서로 정렬되는지
sort_indices = np.argsort(scores)
print(f"정렬 순서: {sort_indices}")
❓ 질문: 표준편차가 크면 뭘 의미할까요?

Step 3: 조건으로 데이터 찾기

scores = np.array([72, 88, 91, 67, 94, 83, 79, 96, 85, 74])

# 90점 이상인 점수들
high_scores = scores[scores >= 90]
print(f"90점 이상: {high_scores}")

# 80점 이상인 학생 수
good_students = np.sum(scores >= 80)
print(f"80점 이상 학생 수: {good_students}")

# 평균보다 높은 점수들
average = np.mean(scores)
above_average = scores[scores > average]
print(f"평균({average:.1f})보다 높은 점수: {above_average}")

# 70점 이상 90점 미만
middle_scores = scores[(scores >= 70) & (scores < 90)]
print(f"70-90점 사이: {middle_scores}")
