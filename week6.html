💻 새로운 그래프들 배우기

Step 1: 막대그래프 (Bar Chart) - 카테고리 비교의 왕

import matplotlib.pyplot as plt
import numpy as np

# 우리 반이 좋아하는 과목 설문조사 결과
subjects = ['수학', '과학', '영어', '국어', '체육', '음악']
votes = [__, __, __, __, __, __]  # 실제 설문 결과 입력

plt.figure(figsize=(10, 6))
bars = plt.bar(subjects, votes, 
               color=['red', 'blue', 'green', 'orange', 'purple', 'pink'],
               alpha=0.7,  # 투명도
               edgecolor='black',  # 테두리 색
               linewidth=1)

plt.title('우리 반이 좋아하는 과목')
plt.xlabel('과목')
plt.ylabel('득표 수')
plt.xticks(rotation=45)  # x축 라벨 45도 회전

# 막대 위에 숫자 표시
for i, v in enumerate(votes):
    plt.text(i, v + 0.1, str(v), ha='center', va='bottom', fontweight='bold')

plt.show()
📊 막대그래프 분석:
  • 1위 과목: (표)
  • 꼴찌 과목: (표)
  • 예상과 다른 결과:

Step 2: 원그래프 (Pie Chart) - 비율의 마법사

# 같은 데이터를 원그래프로

Step 3: 히스토그램 (Histogram) - 분포의 탐정

# 우리 반 키 분포 분석
np.random.seed(42)  # 같은 결과를 위해
heights = np.random.normal(165, 8, 30)  # 평균 165cm, 표준편차 8, 30명

plt.figure(figsize=(10, 6))
n, bins, patches = plt.hist(heights, 
                           bins=8,           # 구간 개수
                           color='skyblue', 
                           alpha=0.7,
                           edgecolor='black',
                           density=False)    # 빈도수로 표시

plt.title('우리 반 키 분포')
plt.xlabel('키 (cm)')
plt.ylabel('학생 수')
plt.grid(True, alpha=0.3)

# 평균선 표시
mean_height = np.mean(heights)
plt.axvline(mean_height, color='red', linestyle='--', 
           label=f'평균: {mean_height:.1f}cm')
plt.legend()
plt.show()

print(f"평균 키: {np.mean(heights):.1f}cm")
print(f"가장 많은 학생이 있는 키 구간: {bins[np.argmax(n)]:.1f}~{bins[np.argmax(n)+1]:.1f}cm")
📏 히스토그램 분석:
  • 평균 키: cm
  • 가장 흔한 키 구간: cm ~ cm
  • 분포 모양:

# 24시간을 어떻게 보내는지 분석
activities = ['수면', '학교', '학원', '숙제', '식사', '여가', '기타']
hours = [__, __, __, __, __, __, __]  # 실제 시간 입력 (합계 24시간)

# 1) 막대그래프로 표현
plt.figure(figsize=(12, 5))

plt.subplot(1, 2, 1)  # 1행 2열 중 첫 번째
plt.bar(activities, hours, color='lightcoral')
plt.title('하루 시간 사용 (막대그래프)')
plt.ylabel('시간')
plt.xticks(rotation=45)

# 2) 원그래프로 표현
plt.subplot(1, 2, 2)  # 1행 2열 중 두 번째
plt.pie(hours, labels=activities, autopct='%1.1f%%', startangle=90)
plt.title('하루 시간 사용 (원그래프)')

plt.tight_layout()  # 자동 레이아웃 조정
plt.show()

# 분석
total_study = hours[1] + hours[2] + hours[3]  # 학교+학원+숙제
print(f"총 공부시간: {total_study}시간 ({total_study/24*100:.1f}%)")
⏰ 시간 사용 패턴 분석:
  • 가장 많은 시간을 보내는 활동: (시간)
  • 총 공부시간: 시간 (%)
  • 개선하고 싶은 점:

미션 2: 월별 용돈 vs 지출 비교

months = ['1월', '2월', '3월', '4월', '5월', '6월']
allowance = [__, __, __, __, __, __]  # 받은 용돈
spending = [__, __, __, __, __, __]   # 쓴 돈

plt.figure(figsize=(10, 6))
x = np.arange(len(months))
width = 0.35

# 나란히 놓인 막대그래프
plt.bar(x - width/2, allowance, width, label='받은 용돈', 
        color='lightblue', alpha=0.8)
plt.bar(x + width/2, spending, width, label='쓴 돈', 
        color='lightcoral', alpha=0.8)

plt.title('월별 용돈 vs 지출')
plt.xlabel('월')
plt.ylabel('금액 (원)')
plt.xticks(x, months)
plt.legend()
plt.grid(True, alpha=0.3)

# 저축 계산 및 표시
savings = np.array(allowance) - np.array(spending)
for i, saving in enumerate(savings):
    if saving >= 0:
        plt.text(i, max(allowance[i], spending[i]) + 1000, 
                f'+{saving:,}', ha='center', color='green', fontweight='bold')
    else:
        plt.text(i, max(allowance[i], spending[i]) + 1000, 
                f'{saving:,}', ha='center', color='red', fontweight='bold')

plt.show()

print(f"총 저축액: {np.sum(savings):,}원")
print(f"평균 월 저축: {np.mean(savings):,.0f}원")
💰 용돈 관리 분석:
  • 가장 많이 저축한 달: 월 (원)
  • 적자가 난 달: 월 (원)
  • 저축 습관 평가:

미션 3: 우리 반 성적 분포 히스토그램

# 가상의 우리 반 시험 점수 (30명)
np.random.seed(123)
scores = np.random.normal(78, 12, 30)  # 평균 78점, 표준편차 12
scores = np.clip(scores, 0, 100)  # 0-100점 범위로 제한

plt.figure(figsize=(12, 8))

# 전체 히스토그램
plt.subplot(2, 2, 1)
plt.hist(scores, bins=8, color='lightgreen', alpha=0.7, edgecolor='black')
plt.title('전체 점수 분포')
plt.xlabel('점수')
plt.ylabel('학생 수')
plt.axvline(np.mean(scores), color='red', linestyle='--', 
           label=f'평균: {np.mean(scores):.1f}점')
plt.legend()

# 등급별 분류
A_grade = scores[scores >= 90]
B_grade = scores[(scores >= 80) & (scores < 90)]
C_grade = scores[(scores >= 70) & (scores < 80)]
D_grade = scores[scores < 70]

# 등급별 학생 수
plt.subplot(2, 2, 2)
grades = ['A급\n(90+)', 'B급\n(80-89)', 'C급\n(70-79)', 'D급\n(70미만)']
counts = [len(A_grade), len(B_grade), len(C_grade), len(D_grade)]
colors = ['gold', 'lightgreen', 'orange', 'lightcoral']

plt.bar(grades, counts, color=colors)
plt.title('등급별 학생 수')
plt.ylabel('학생 수')

# 등급별 비율 (원그래프)
plt.subplot(2, 2, 3)
plt.pie(counts, labels=grades, autopct='%1.1f%%', colors=colors, startangle=90)
plt.title('등급별 비율')

# 점수 구간별 상세 분석
plt.subplot(2, 2, 4)
bins = [0, 60, 70, 80, 90, 100]
plt.hist(scores, bins=bins, color='skyblue', alpha=0.7, edgecolor='black')
plt.title('점수 구간별 분포')
plt.xlabel('점수 구간')
plt.ylabel('학생 수')
plt.xticks([60, 70, 80, 90, 100])

plt.tight_layout()
plt.show()

# 통계 출력
print("=== 우리 반 성적 분석 ===")
print(f"평균 점수: {np.mean(scores):.1f}점")
print(f"최고 점수: {np.max(scores):.1f}점")
print(f"최저 점수: {np.min(scores):.1f}점")
print(f"A급 학생: {len(A_grade)}명 ({len(A_grade)/len(scores)*100:.1f}%)")
print(f"평균 이상 학생: {np.sum(scores >= np.mean(scores))}명")
📈 성적 분포 분석:
  • 우리 반 평균: 점
  • A급 비율: % (명)
  • 가장 많은 학생이 속한 구간:
  • 분포의 특징:
