오늘은 실제 데이터로 NumPy의 모든 기능을 활용해보는 날입니다!

📊 데이터 준비

import numpy as np

# 우리 반 10명의 신체 정보 (가상 데이터)
names = ['김철수', '이영희', '박민수', '최지영', '정태현', 
         '한소영', '윤준호', '장미래', '오성민', '배하늘']

heights = np.array([165, 172, 158, 168, 175, 162, 180, 169, 173, 167])  # 키(cm)
weights = np.array([55, 68, 52, 60, 70, 58, 75, 62, 69, 63])           # 몸무게(kg)
ages = np.array([14, 15, 14, 15, 14, 15, 14, 15, 14, 15])              # 나이

print("=== 우리 반 신체 정보 ===")
print(f"학생 수: {len(heights)}명")

💻 분석 실습

1단계: 기본 통계 계산

print("=== 키 분석 ===")
print(f"평균 키: {np.mean(heights):.1f}cm")
print(f"최대 키: {np.max(heights)}cm") 
print(f"최소 키: {np.min(heights)}cm")
print(f"키 표준편차: {np.std(heights):.1f}cm")

print("\n=== 몸무게 분석 ===")
print(f"평균 몸무게: {np.mean(weights):.1f}kg")
print(f"최대 몸무게: {np.max(weights)}kg")
print(f"최소 몸무게: {np.min(weights)}kg")
print(f"몸무게 표준편차: {np.std(weights):.1f}kg")
📊 내 계산 결과:
평균 키: cm
키 범위: cm ~ cm
평균 몸무게: kg
몸무게 범위: kg ~ kg

2단계: BMI 계산하기

# BMI = 몸무게(kg) / (키(m))²
bmi = weights / ((heights/100) ** 2)

print("=== BMI 분석 ===")
print(f"평균 BMI: {np.mean(bmi):.1f}")
print(f"최고 BMI: {np.max(bmi):.1f}")
print(f"최저 BMI: {np.min(bmi):.1f}")

# BMI 상태 분석
underweight = np.sum(bmi < 18.5)  # 저체중
normal = np.sum((bmi >= 18.5) & (bmi < 23))  # 정상
overweight = np.sum(bmi >= 23)  # 과체중

print(f"저체중: {underweight}명")
print(f"정상체중: {normal}명") 
print(f"과체중: {overweight}명")
📊 BMI 기록:
평균 BMI:
저체중: 명, 정상: 명, 과체중: 명

3단계: 키 순서대로 정렬하기

# 키 순서로 정렬하기 위한 인덱스
height_order = np.argsort(heights)

print("=== 키 순서 (작은 순) ===")
for i, idx in enumerate(height_order):
    print(f"{i+1}등: {names[idx]} - {heights[idx]}cm, {weights[idx]}kg")

# 키 큰 순서
reverse_order = np.argsort(heights)[::-1]

print("\n=== 키 순서 (큰 순) ===")  
for i, idx in enumerate(reverse_order[:3]):  # 상위 3명만
    print(f"{i+1}등: {names[idx]} - {heights[idx]}cm")
🏆 순위 기록:
1등 (가장 키 큰 사람):
2등:
3등:

4단계: 조건별 분석

# 키 170cm 이상인 학생들
tall_students = heights >= 170
print("=== 키 170cm 이상 학생들 ===")
for i, is_tall in enumerate(tall_students):
    if is_tall:
        print(f"{names[i]}: {heights[i]}cm")

print(f"총 {np.sum(tall_students)}명")

# 몸무게가 평균보다 가벼운 학생들
avg_weight = np.mean(weights)
light_students = weights < avg_weight

print(f"\n=== 평균({avg_weight:.1f}kg)보다 가벼운 학생들 ===")
for i, is_light in enumerate(light_students):
    if is_light:
        print(f"{names[i]}: {weights[i]}kg")
📋 조건별 결과:
키 170cm 이상: 명
평균보다 가벼운 학생: 명

🔍 심화 분석

연령별 분석

# 14세와 15세 그룹 비교
age_14 = ages == 14
age_15 = ages == 15

print("=== 연령별 비교 ===")
print(f"14세 평균 키: {np.mean(heights[age_14]):.1f}cm")
print(f"15세 평균 키: {np.mean(heights[age_15]):.1f}cm")
print(f"14세 평균 몸무게: {np.mean(weights[age_14]):.1f}kg")
print(f"15세 평균 몸무게: {np.mean(weights[age_15]):.1f}kg")

상관관계 분석 (심화)

# 키와 몸무게의 관계
correlation = np.corrcoef(heights, weights)[0, 1]
print(f"키와 몸무게의 상관관계: {correlation:.3f}")

# 상관관계 해석
if correlation > 0.7:
    print("강한 양의 상관관계 - 키가 클수록 몸무게도 많이 나간다")
elif correlation > 0.3:
    print("약한 양의 상관관계 - 키와 몸무게가 어느 정도 관련있다")
else:
    print("상관관계가 약하다")
