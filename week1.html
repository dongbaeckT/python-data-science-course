Step 3: 리스트 vs 배열 비교하기

# 리스트로 계산
list_data = [1, 2, 3, 4, 5]
list_result = [x * 2 for x in list_data]
print("리스트 결과:", list_result)

# 배열로 계산  
array_data = np.array([1, 2, 3, 4, 5])
array_result = array_data * 2
print("배열 결과:", array_result)
